(

CREATE TABLE coments
(
	com_id               INTEGER NOT NULL,

	text                 VARCHAR(200) NULL

CREATE TABLE news
(
	news_id              INTEGER NOT NULL,
	title                VARCHAR(20) NULL,
	text                 VARCHAR(500) NULL,
	cat_id               INTEGER NULL
);

ALTER TABLE news
ADD PRIMARY KEY (news_id);

(
	news_id              INTEGER NOT NULL,
	us_id                INTEGER NOT NULL

(
	us_id                INTEGER NOT NULL,
